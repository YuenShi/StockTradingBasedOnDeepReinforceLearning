def get_data(symbol=['AAPL'], startdate = "2015-01-01", enddate = "2018-08-25", download = True):
    #   
    _symbol = symbol #AAPL
    symbol_len = len(_symbol)
    # 
    _startdate = startdate
    # 
    _enddate = enddate
    # 
    _download = download
    
    # 
    data_dict = dict()
    
    # symbol
    market_list = ['^GSPC', '^RUT', '^VIX', '^TNX', '^IRX', 'GLD', 'USO','WFC']
    # name -- key
    name_list = ['sp500','rus2000','vix','_10yr_bond','_13wk_bill', 'gold','oil','WFC']
    for i in range(symbol_len):
        # symbol
        market_list.insert(i, _symbol[i])
        # keysymbol
        name_list.insert(i,_symbol[i])
    
    market_len = len(market_list)
    name_len = len(name_list)
    
    if market_len != name_len:
        print("Market-List dosen't match Name-List!")
        return None
    
    # 
    if _download == True:
        try:
            for i in range(market_len):
                # namesymbol
                n = name_list[i]
                s = market_list[i]
                # 
                data = pdr.get_data_yahoo(n, start = _startdate, end = _enddate)
                data.to_csv("{}.csv".format(s))
        # 
        except IOError as e:
            print(e)
            
    # 
    try:
        for i in range(name_len):
            # name
            n = name_list[i]
            
            # 
            data = pd.read_csv("{}.csv".format(n))
            data_dict[n] = data
        return data_dict
    except IOError as e:
        print(e)
        return None

import numpy as np
import random
from collections import deque
from keras.models import Sequential
from keras.layers import Dense, Dropout, Activation
from keras.optimizers import Adam, SGD


class DQN:
    def __init__(self, env):
        # env
        self.env = env
        # 
        self.memory  = deque(maxlen=500)  # 2000
        # 
        self.gamma = 0.9  
        #  
        self.epsilon = 1.0
        # 
        self.epsilon_min = 0.01 
        # 
        self.epsilon_decay = 0.995
        # 
        self.learning_rate = 0.001
        # 
        self.tau = 0.05
        #
        self.batch_size = 32

        #  eval_model
        self.eval_model = self.create_model()
        #  target_model
        self.target_model = self.create_model()

    def create_model(self):

#         # 
#         model = Sequential()

#         # 32
#         model.add(Dense(32, input_dim=1, activation="relu"))
#         model.add(Dropout(0.5))

#         # 64
#         model.add(Dense(64, activation="relu"))
#         model.add(Dropout(0.5))

#         # 32
#         model.add(Dense(32, activation="relu"))
#         model.add(Dropout(0.5))

#         # 
#         model.add(Dense(self.env.n_actions, activation="softmax")) # linear

#         # 
#         model.compile(loss="mean_squared_error", optimizer=Adam(lr=self.learning_rate))

       
        model = Sequential()
         # 32 features 
        model.add(Dense(32, input_dim=1,activation='relu'))
        model.add(Dropout(0.5))
        #    = (+ max(,))/2
        model.add(Dense(16, activation='relu'))
        model.add(Dropout(0.5))
        # 3 outputs
        model.add(Dense(self.env.n_actions, activation='softmax'))
        # losssparse_categorical_crossentropcategorical_crossentropyoptimizer=Adam(lr=self.learning_rate)
        model.compile(loss='categorical_crossentropy', optimizer=SGD(lr=self.learning_rate),metrics=['accuracy'])
    
        return model

    def act(self, state):
        # 
        self.epsilon *= self.epsilon_decay
        #  epsilon  epsilon_min 
        self.epsilon = max(self.epsilon_min, self.epsilon)
        # 
        if np.random.random() < self.epsilon:
            return np.random.randint(0, self.env.n_actions)
        #  eval_model 
        return np.argmax(self.eval_model.predict(state)[0])

    def remember(self, state, action, reward, new_state, done):
        # 
        self.memory.append([state, action, reward, new_state, done])

    def replay(self):
        # 
        batch_size = self.batch_size
        # 
        if len(self.memory) < batch_size: 
            # 
            return
        # 
        # memorybatch_size
        samples = random.sample(self.memory, batch_size)
        # 
        for sample in samples:
            # 
            state, action, reward, new_state, done = sample
            # target_modeltarget
            target = self.target_model.predict(state)
            # 
            if done:
                # reward 
                target[0][action] = reward
            else:
                # rewardQ_future
                Q_future = max(self.target_model.predict(new_state)[0])
                # Q_futurereward
                target[0][action] = reward + Q_future * self.gamma
            # statetargeteval_model
            self.eval_model.fit(state, target, epochs=1, verbose=0)

    def target_train(self):
        # eval_model
        eval_weights = self.eval_model.get_weights()
        # target_model
        target_weights = self.target_model.get_weights()
        # target_model
        for i in range(len(target_weights)):
            # taueval_modeltarget_modeltarget_model
            target_weights[i] = eval_weights[i] * self.tau + target_weights[i] * (1 - self.tau)
        # target_model
        self.target_model.set_weights(target_weights)
        
    def save_model(self, fn):
        # eval_model
        self.eval_model.save(fn)

class Runner:
    def __init__(self):
        # model
        K.clear_session()
        tf.reset_default_graph()
        
        # 
        self.gamma = 0.9
        # 
        self.epsilon = 1.0
        # 
        self.epsilon_min = 0.01 
        # 
        self.epsilon_decay = 0.995  
        # 
        self.train_success = False
        # runner
        self.model_name = ''
        
    def trainer(self, symbol, env, epochs):
        # 
        self.env = env
        # dqn_agent
        self.dqn_agent = DQN(env=self.env)
        # 
        self.epochs = epochs
        # 
        self.epoch_len = self.env.sample_size

        # epochs
        for epoch in range(self.epochs):
            # 
            self.env.reset()
            # 
            cur_state = self.env.states
            # 
            fortune = list()
            # 
            stock = list()
            # 
            cash = list()
            # 
            act = list()
            # 
            re = list()
            # 
            for step in range(self.epoch_len):
                # action
                action = self.dqn_agent.act(cur_state)
                # action
                new_state, done, reward = self.env.execute(action)

                # action
                act.append(action)
                # reward
                re.append(reward)
                # stock
                stock.append(new_state[-3])
                # cash
                cash.append(new_state[-2])
                # fortune
                fortune.append(new_state[-1])

                # 
                self.dqn_agent.remember(cur_state, action, reward, new_state, done)
                # 
                self.dqn_agent.replay() 
                if step > 20:
                    # 
                    self.dqn_agent.target_train() 
                # 
                cur_state = new_state

                # 
                if done:
                    # success
                    if fortune[-1] >= 130000.:
                        self.train_success = True
                        self.model_name = "success-model-{}-{}.h5".format(symbol, epoch)
                        self.dqn_agent.save_model(self.model_name)
                    # train
                    else:
                        self.train_success = False
                        self.model_name = "train-model-{}-{}.h5".format(symbol, epoch)
